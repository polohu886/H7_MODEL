```c
#include "AD9833.h"

/* 仅需定义一次引脚配置 */
driverIO AD9833_FSYNC = {GPIOC, GPIO_PIN_2};
driverIO AD9833_SCLK = {GPIOC, GPIO_PIN_1};
driverIO AD9833_SDATA = {GPIOC, GPIO_PIN_0};
//导入main.h就行

void System_Init(void)
{
    // 1. 在MX_GPIO_Init()中初始化GPIO时钟和引脚模式（推挽输出）
    
    // 2. 初始化AD9833 (仅需设置参考时钟)
    AD9833_Init(25000000);  // 25MHz参考时钟
    
    // 3. 生成1kHz正弦波
    AD9833_SetFrequencyQuick(1000.0f, AD9833_WAVEFORM_SINE);
    
    // 4. 高级示例: 5kHz方波 + 90度相位偏移
    AD9833_SetFrequency(AD9833_FREQ_REG_1, 5000.0f, AD9833_WAVEFORM_SQUARE);
    AD9833_SetPhase(AD9833_PHASE_REG_1, 90.0f);
    AD9833_Setup(AD9833_FREQ_REG_1, AD9833_PHASE_REG_1, AD9833_WAVEFORM_SQUARE);
}
```
